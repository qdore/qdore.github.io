随机数 random


随机数 random

#include <random>
#include <iostream>
using namespace std;
int main()

{
    random_device rd;
    uniform_int_distribution<int> u(0, 9);            //其它分布如正态分布
    default_random_engine e(rd());                          //normal_distribution<>,0-1分布 bernoulli_distribution b(0.5) 0.5表示为1的概率
    for(int i = 1; i < 10; i++)
    {
        cout << u(e) << endl;
    }
    
    default_random_engine e1(2), e2(2);               //种子相同时，序列也相同！
    for(int i = 1; i < 10; i++)
    {
        cout << e1() << ":" << e2() << endl;
    }
}

bitset


bitset



bitset<2> b(2); //初始化bitset

b.any(); //是否存在置位的进制位

b.all(); //所有的位是否都置位了

b.none(); //不存在置位的进制位码

b.count(); //置位的个数

b.size(); //b的位数

b.test(2); //第2位是否为1

b.set(2,1); //设置第二位为1

b.reset(2); //重置第二位

b.reset(); //重置所有位

b[1]; //访问第二个

b.to_ulong(); //转变类型

b.to_ullong();

b.to_string();
