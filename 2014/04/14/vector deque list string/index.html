int main(){

string s = "woxianzaiyoudianshi";

s.erase(2, 3); //s.erase(2)是删除2开始的所有元素

s.insert(2, "oxian"); //"oxian"可以用迭代器替换

s.replace(2, 3, "shf");



deque<int> c;

 int a = 0;

 int b[] = {0,1,2};

deque<int>::iterator p = c.begin();

c.push_back(a); //尾部插入a

c.push_front(a); //首部插入a

c.insert(p, a); //在p之前插入a

c.insert(p, 2, a); //在p之前插入两个a

c.insert(p, begin(b), end(b)); //在p之前插入b

c.insert(p, {0,1});



c.pop_back(); //尾部推出，返回void

c.pop_front(); //首部推出，返回void

c.erase(p); //删除p

c.erase(c.begin(), c.end()); //删除c

c.clear(); //删除c



c.resize(10); //调整c的大小为10个元素，多出得丢弃

c.resize(25, 0); //调整c的大小为10个元素，默认为0



forward_list<int> lst; //list是双向链表

forward_list<int>::iterator q = lst.before_begin(); //返回链表首元素的前一个元素

lst.insert_after(q, a); //在q之后插入a

lst.insert_after(q, 10, a); //在q之后插入10个a

lst.insert_after(q,{0,1});



lst.erase_after(q); //删除q之后的元素

lst.erase_after(q,lst.end()); //删除q之后知道lst.end(),不包括lst.end()

}
